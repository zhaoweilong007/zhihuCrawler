<span class="kn">import</span> <span class="nn">random</span>

<span class="n">FORCE_UNIVERSAL_ANSWER_RATE</span> <span class="o">=</span> <span class="mf">0.13</span>

<span class="n">UNIVERSAL_ANSWERS</span> <span class="o">=</span> <span class="p">[</span>
        <span class="s2">"你说你马呢？"</span><span class="p">,</span>
        <span class="s2">"那没事了。"</span><span class="p">,</span>
        <span class="s2">"真别逗我笑啊。"</span><span class="p">,</span>
        <span class="s2">"那可真是有趣呢。"</span><span class="p">,</span>
        <span class="s2">"就这？就这？"</span><span class="p">,</span>
        <span class="s2">"你品，你细品。"</span><span class="p">,</span>
        <span class="s2">"不会真有人觉得是这样的吧，不会吧不会吧？"</span><span class="p">,</span>
        <span class="s2">"你在教我做事？"</span>
<span class="p">]</span>

<span class="n">STRONG_EMOTION_ANSWERS</span> <span class="o">=</span> <span class="p">[</span>
        <span class="s2">"你急了急了急了？"</span><span class="p">,</span>
        <span class="s2">"他急了，他急了！"</span>
<span class="p">]</span>

<span class="n">QUESTION_ANSWERS</span> <span class="o">=</span> <span class="p">[</span>
        <span class="s2">"不会真有人还不知道吧？"</span><span class="p">,</span>
        <span class="s2">"你都不知道，那你说你马呢？"</span>
<span class="p">]</span>

<span class="n">STRONG_EMOTION_PATTERNS</span> <span class="o">=</span> <span class="p">[</span>
        <span class="s2">"！"</span><span class="p">,</span>
        <span class="s2">"？？？"</span><span class="p">,</span>
        <span class="s2">"???"</span><span class="p">,</span>
        <span class="s2">"气抖冷"</span>
<span class="p">]</span>

<span class="n">QUESTION_PATTERNS</span> <span class="o">=</span> <span class="p">[</span>
        <span class="s2">"？"</span><span class="p">,</span>
        <span class="s2">"怎么"</span><span class="p">,</span>
        <span class="s2">"什么"</span><span class="p">,</span>
        <span class="s2">"咋"</span><span class="p">,</span>
        <span class="s2">"?"</span>
<span class="p">]</span>

<span class="k">def</span> <span class="nf">check_patterns</span><span class="p">(</span><span class="n">str_in</span><span class="p">,</span> <span class="n">patterns</span><span class="p">):</span>
    <span class="k">for</span> <span class="n">p</span> <span class="ow">in</span> <span class="n">patterns</span><span class="p">:</span>
        <span class="k">if</span> <span class="n">p</span> <span class="ow">in</span> <span class="n">str_in</span><span class="p">:</span>
            <span class="k">return</span> <span class="kc">True</span>
    <span class="k">return</span> <span class="kc">False</span>


<span class="k">def</span> <span class="nf">pick_answer</span><span class="p">(</span><span class="n">question</span><span class="p">):</span>
    <span class="k">if</span> <span class="n">random</span><span class="o">.</span><span class="n">random</span><span class="p">()</span> <span class="o">&lt;</span> <span class="n">FORCE_UNIVERSAL_ANSWER_RATE</span><span class="p">:</span>
        <span class="k">return</span> <span class="n">random</span><span class="o">.</span><span class="n">choice</span><span class="p">(</span><span class="n">UNIVERSAL_ANSWERS</span><span class="p">)</span>

    <span class="k">if</span> <span class="n">check_patterns</span><span class="p">(</span><span class="n">question</span><span class="p">,</span> <span class="n">STRONG_EMOTION_PATTERNS</span><span class="p">):</span>
        <span class="k">return</span> <span class="n">random</span><span class="o">.</span><span class="n">choice</span><span class="p">(</span><span class="n">STRONG_EMOTION_ANSWERS</span><span class="p">)</span>
    <span class="k">elif</span> <span class="n">check_patterns</span><span class="p">(</span><span class="n">question</span><span class="p">,</span> <span class="n">QUESTION_PATTERNS</span><span class="p">):</span>
        <span class="k">return</span> <span class="n">random</span><span class="o">.</span><span class="n">choice</span><span class="p">(</span><span class="n">QUESTION_ANSWERS</span><span class="p">)</span>
    <span class="k">else</span><span class="p">:</span>
        <span class="k">return</span> <span class="n">random</span><span class="o">.</span><span class="n">choice</span><span class="p">(</span><span class="n">UNIVERSAL_ANSWERS</span><span class="p">)</span>

<span class="k">def</span> <span class="nf">main</span><span class="p">():</span>
    <span class="k">while</span> <span class="kc">True</span><span class="p">:</span>
        <span class="n">question</span> <span class="o">=</span> <span class="nb">input</span><span class="p">(</span><span class="s2">"&lt;&lt;&lt; "</span><span class="p">)</span>
        <span class="nb">print</span><span class="p">(</span><span class="s2">"&gt;&gt;&gt; "</span> <span class="o">+</span> <span class="n">pick_answer</span><span class="p">(</span><span class="n">question</span><span class="p">))</span>

<span class="k">if</span> <span class="vm">__name__</span> <span class="o">==</span> <span class="s2">"__main__"</span><span class="p">:</span>
    <span class="n">main</span><span class="p">()</span></code></pre>
