 <p data-pid="Ei2zS30L">纯色头像了解一下</p>
 <div class="highlight">
  <pre><code class="language-text">import cv2
import numpy as np

def generate_img(color):
    img = np.ones((128, 128, 1)) * [int(color[4:6], 16), int(color[2:4], 16), int(color[0:2], 16)]
    return img

cv2.imwrite('纯黑.png', generate_img("000000"))</code></pre>
